## this patch turns all protein atoms ito virtual atoms
## useful if you want the residues in the pose as placeholders...
NAME Virtual_Protein_Residue
TYPES    VIRTUAL_RESIDUE_VARIANT

BEGIN_SELECTOR
PROPERTY PROTEIN
NOT VARIANT_TYPE UPPER_TERMINUS_VARIANT
NOT VARIANT_TYPE LOWER_TERMINUS_VARIANT
NOT VARIANT_TYPE CUTPOINT_UPPER
NOT VARIANT_TYPE CUTPOINT_LOWER
# pH RTs change atom counts while retaining aa etc.
# until we reimplement these patches as "virtualize this set of atoms" generally
# they can't have this patch.
NOT VARIANT_TYPE DEPROTONATED
NOT VARIANT_TYPE PROTONATED
NOT PROPERTY BRANCH_POINT
NOT PROPERTY PHOSPHONATE
# only standard AA's defined for now:
AA ALA ARG ASN ASP CYS CYS CYS GLN GLU GLY HIS HIS ILE LEU LYS MET PHE PRO SER THR TRP TYR VAL
END_SELECTOR

BEGIN_CASE ###  ALA ###

BEGIN_SELECTOR
AA ALA
END_SELECTOR
SET_ATOM_TYPE  N    VIRT
SET_ATOM_TYPE  CA   VIRT
SET_ATOM_TYPE  C    VIRT
SET_ATOM_TYPE  O    VIRT
SET_ATOM_TYPE  CB   VIRT
SET_ATOM_TYPE  H    VIRT
SET_ATOM_TYPE  HA   VIRT
SET_ATOM_TYPE 1HB   VIRT
SET_ATOM_TYPE 2HB   VIRT
SET_ATOM_TYPE 3HB   VIRT
ADD_PROPERTY VIRTUAL_RESIDUE
END_CASE
