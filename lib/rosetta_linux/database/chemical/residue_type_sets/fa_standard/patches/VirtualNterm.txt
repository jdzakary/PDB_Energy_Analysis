## this patch turns N-terminus virtual
NAME VirtualNterm
TYPES VIRTUAL_NTERM

BEGIN_SELECTOR
PROPERTY PROTEIN
VARIANT_TYPE LOWER_TERMINUS_VARIANT
NOT VARIANT_TYPE METHYLATED_NTERM_VARIANT
NOT VARIANT_TYPE N_FORMYLATION
NOT VARIANT_TYPE OOP_PRE
END_SELECTOR

BEGIN_CASE ### GLY
BEGIN_SELECTOR
AA GLY
END_SELECTOR
SET_ATOM_TYPE  N VIRT
SET_ATOM_TYPE 1H VIRT
SET_ATOM_TYPE 2H VIRT
SET_ATOM_TYPE 3H VIRT
SET_ATOM_TYPE 2HA VIRT # Why does this count?
END_CASE

END_SELECTOR
